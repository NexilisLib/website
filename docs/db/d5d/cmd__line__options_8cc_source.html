83CmdLineOptions::CmdLineOptions(int argc, char** argv)
84{
85 // Basic command line parsing
86 for (int i = 1; i < argc; ++i)
87 {
88 std::string arg = argv[i];
89 if (arg[0] == '-')
90 {
91 std::string name = arg;
92 std::vector<std::shared_ptr<IValue>> values;
93
94 // Collect all following non-option arguments as values
95 while (i + 1 < argc && argv[i + 1][0] != '-')
96 {
97 std::string val = argv[++i];
98
99 // Try to parse as different types
100 if (int intValue; tryParse(val, intValue))
101 {
102 values.emplace_back(std::make_shared<Value<int>>(intValue));
103 }
104 else if (double doubleValue; tryParse(val, doubleValue))
105 {
106 values.emplace_back(std::make_shared<Value<double>>(doubleValue));
107 }
108 else if (std::string stringValue; tryParse(val, stringValue))
109 {
110 values.emplace_back(std::make_shared<Value<std::string>>(val));
111 }
112 else
113 {
114 Log::error("Undefined value as command line option!");
115 }
116 }
117 m_arguments.emplace_back(name, values);
118 }
119 }
120}
