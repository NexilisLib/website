176nx_data Packet::clientIdentification()
177{
178 if (!m_clientApi)
179 {
180 Log::error("Packet has not initialized ClientAPI");
181 return {};
182 }
183
184 uint64_t clientId = m_clientApi->getClientId();
185 if (clientId == 0)
186 {
187 Log::error("Error creating new message id");
188 return {};
189 }
190
191 auto clientIdVector = Util::convertToByteVector(clientId);
192
193 assert(!clientIdVector.empty());
194 assert(clientIdVector.size() == 8);
195 assert(Util::convertToType<uint64_t>(clientIdVector) != 0);
196
197 auto messageIdVector = Util::convertToByteVector(m_clientApi->getNewMessageId());
198 assert(messageIdVector.size() == 8);
199
200 clientIdVector.reserve(clientIdVector.size() + messageIdVector.size());
201 std::copy(messageIdVector.begin(), messageIdVector.end(), std::back_inserter(clientIdVector));
202
203 return clientIdVector;
204}
