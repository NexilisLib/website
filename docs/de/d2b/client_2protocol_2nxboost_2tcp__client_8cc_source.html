137{

139
140 boost::asio::streambuf receiveBuffer;
141 boost::system::error_code error;
142
143 boost::asio::read(m_socket, receiveBuffer, boost::asio::transfer_at_least(1), error);
144
145 // Handle the case where the socket is closed or an error occurs
146 if (error == boost::asio::error::operation_aborted || error == boost::asio::error::eof)
147 {
148 return false;
149 }
150
151 if (error)
152 {
153 Log::error("Receive error: " + error.message());
154 return false; // Some other error occurred
155 }
156
157 uint64_t size = receiveBuffer.size();
158 if (size == 0)
159 {
160 return false;
161 }
162
163 // Allocate enough space in the buffer vector and copy data.
164 buffer.resize(size);
165 std::istream is(&receiveBuffer);
166 is.read(reinterpret_cast<char*>(buffer.data()), size);
167
168 return true;
