216uint64_t Util::uint64FromFront(const nx_data& vec)
217{
218 if (vec.size() < 8)
219 {
220 // Not enough data to read 8 bytes
221 Log::error("Not enough data to read 8 bytes");
222 return 0;
223 }
224
225 uint64_t value = 0;
226
227 if (server::Config::getBigEndian())
228 {
229 // Big-endian: Most significant byte is at the lowest address
230 for (int i = 0; i < 8; ++i)
231 {
232 value |= static_cast<uint64_t>(vec[i]) << ((7 - i) * 8);
233 }
234 }
235 else
236 {
237 // Little-endian: Least significant byte is at the lowest address
238 for (int i = 0; i < 8; ++i)
239 {
240 value |= static_cast<uint64_t>(vec[i]) << (i * 8);
241 }
242 }
243 return value;
244}
